说明

将一个货品添加到购物车

方法

/api/mobile/order!storeCartGoods.do

参数说明备注

无无需要会员登录

示例代码

var url = getFullUrl('/api/mobile/order!storeCartGoods.do'); var options = { url : url, success : function(data) { if (data.result == 1) { var viewData = data.data; var price = viewData.order_price.needPayMoney; $("#price_p").html("总计：" + e$.formatMoney(price)); initTpl(viewData.store_list, $("#goods_html_templet"), $("#goods_html_view"), closeLoad); } else { e$.alert("获取资源失败，请稍后重试。"); } } }; e$.ajax(options);
调用结果：
{ "result": 1, "data": { "store_list": [], "order_price": { "goodsPrice": 0, "orderPrice": 0, "shippingPrice": 0, "needPayMoney": 0, "discountPrice": 0, "weight": 0, "point": 0, "discountItem": {} } } }
