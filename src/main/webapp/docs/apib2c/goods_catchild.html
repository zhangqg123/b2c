说明

获取商品分类子分类列表

方法

/api/mobile/goodscat!goodsCatList.do

参数说明备注

parentid父分类IDint型，父分类ID可为空

示例代码

var param = e$.getParam(); var url = getFullUrl('/api/mobile/goodscat!goodsCatList.do'); var options = { url : url, data : {parentid:param.catid}, success : function(data) { if (data.result == 1) { var viewData = data.data; classifyData = viewData; initTpl(viewData, $("#classify_html_templet"), $("#classify_html_view"), bind); } else { e$.alert("获取资源失败，请稍后重试。"); } } }; e$.ajax(options);
