API

/api/mobile/address/default-address.do

参数说明备注

无无此api需要会员登录

var url = getFullUrl('/api/mobile/address/default-address.do'); var options = { url : url, success : function(data) { if (data.result == 1) { alert(JSON.stringify(data)); } else { e$.alert("删除失败，请稍后重试。"); } } }; e$.ajax(options);
